与腾讯微信的整合

DOMReady原来是等于页面节点建完之后就触发，现在还需要等到微信SDK加载完毕才可以用，它提供了一个wx.ready。

var oldReady = avalon.ready
avalon.ready = function(fn) {
    wx.ready(function() {
        oldReady(fn)
    })
}

与微信脚本混用,请确保 其sdk放在avalon的前面

<script src='http://res.wx.qq.com/open/js/jweixin-1.0.0.js'>
</script>
<script src='avalon.js'></script>
<script src='main.js'></script>
<!--main.js为业务代码-->
